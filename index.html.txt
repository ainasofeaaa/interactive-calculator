        function compileExpression(exprStr) {
            try {
                // Pre-clean standard inputs

                // Allow implicit multiplications (e.g. 2x instead of 2*x)
                const parsed = math.compile(cleanStr);
                
                // Verify evaluation with dummy value

                if (typeof test !== 'number' && (typeof test !== 'object' || !test.isComplex)) {
                    throw new Error("Invalid output dimension");

                compiledExpr = null;
                document.getElementById('function-input').classList.add('border-rose-500');
                const errElement = document.getElementById('function-error');
                const errMsg = document.getElementById('error-message');
                errMsg.innerText = `${e.message}`;
                errElement.classList.remove('hidden');
                return false;
            }
        }

        // Safely evaluate f(x) for real values
        function evalF(x) {
            if (!compiledExpr) return NaN;
            try {
                let val = compiledExpr.evaluate({ x: x });
                if (typeof val === 'object' && val.isComplex) {
                    if (val.im === 0) return val.re;
                    return NaN; // Filter imaginary results
                }
                if (typeof val === 'number') {
                    if (isFinite(val)) return val;
                }
                return NaN;
            } catch (e) {
                return NaN;
            }
        }

        // INTEGRATION SIMULATION ENGINE (High resolution Simpson's Rule for the "Exact" Value)
        function calculateExactIntegral(a, b, numSteps = 1000) {
            const h = (b - a) / numSteps;
            let sumSigned = 0;
            let sumAbsolute = 0;

            for (let i = 0; i <= numSteps; i++) {
                const x = a + i * h;
                const y = evalF(x);
                if (isNaN(y) || !isFinite(y)) continue;

                // Trapezoidal Weights
                const weight = (i === 0 || i === numSteps) ? 0.5 : 1.0;
                sumSigned += weight * y;
                sumAbsolute += weight * Math.abs(y);
            }

            return {
                signed: sumSigned * h,
                absolute: sumAbsolute * h
            };
        }

        // COMPUTE APPROXIMATE RIEMANN SUM VALUES
        function calculateApproximateIntegral(a, b, method, n) {
            const h = (b - a) / n;
            let sum = 0;

            for (let i = 0; i < n; i++) {
                const x_start = a + i * h;
                const x_end = a + (i + 1) * h;
                
                let evalPoint = x_start; // default left
                if (method === 'right') {
                    evalPoint = x_end;
                } else if (method === 'midpoint') {
                    evalPoint = x_start + h / 2;
                } else if (method === 'trapezoid') {
                    const y1 = evalF(x_start);
                    const y2 = evalF(x_end);
                    if (isFinite(y1) && isFinite(y2)) {
                        sum += 0.5 * (y1 + y2) * h;
                    }
                    continue;
                }
                
                const y = evalF(evalPoint);
                if (isFinite(y)) {
                    sum += y * h;
                }
            }
            return sum;
        }

        // RE-CALCULATE AND UPDATE VALUE METRICS
        function recalculateAndRender() {
            if (!compiledExpr) return;

            // Perform Calculations
            const exact = calculateExactIntegral(limitA, limitB);
            
            // Display exacts
            document.getElementById('result-signed-area').innerText = exact.signed.toFixed(5);
            document.getElementById('result-absolute-area').innerText = exact.absolute.toFixed(5);
            
            const errorCard = document.getElementById('numerical-error-card');
            
            if (approxMethod !== 'exact') {
                const approxVal = calculateApproximateIntegral(limitA, limitB, approxMethod, subdivisionsN);
                const errValue = approxVal - exact.signed;
                let pctError = 0;
                
                if (Math.abs(exact.signed) > 1e-9) {
                    pctError = (Math.abs(errValue) / Math.abs(exact.signed)) * 100;
                } else if (Math.abs(approxVal) > 1e-9) {
                    pctError = 100.0;
                }

                errorCard.classList.remove('hidden');
                document.getElementById('result-error-value').innerText = (errValue >= 0 ? "+" : "") + errValue.toFixed(5);
                document.getElementById('result-error-percent').innerText = pctError.toFixed(2) + "% Error";
                
                // Show updated banner badge
                document.getElementById('hud-approximation-badge').classList.remove('hidden');
                document.getElementById('hud-approximation-badge').classList.add('flex');
                
                let methodLabel = "Riemann Sum";
                if (approxMethod === 'left') methodLabel = "Left Riemann Sum";
                if (approxMethod === 'right') methodLabel = "Right Riemann Sum";
                if (approxMethod === 'midpoint') methodLabel = "Midpoint Riemann Sum";
                if (approxMethod === 'trapezoid') methodLabel = "Trapezoidal Approximation";
                document.getElementById('hud-approx-method-label').innerText = methodLabel;
            } else {
                errorCard.classList.add('hidden');
                document.getElementById('hud-approximation-badge').classList.add('hidden');
                document.getElementById('hud-approximation-badge').classList.remove('flex');
            }

            // Update LaTeX visual Formula
            try {
                const node = math.parse(fExpression);
                const latex = node.toTex();
                
                let integralTexVal = "";
                if (approxMethod === 'exact') {
                    integralTexVal = `\\int_{${limitA.toFixed(2)}}^{${limitB.toFixed(2)}} f(x)\\,dx = ${exact.signed.toFixed(4)}`;
                } else {
                    const approxVal = calculateApproximateIntegral(limitA, limitB, approxMethod, subdivisionsN);
                    integralTexVal = `\\int_{${limitA.toFixed(2)}}^{${limitB.toFixed(2)}} f(x)\\,dx \\approx ${approxVal.toFixed(4)}`;
                }
                
                katex.render(integralTexVal, document.getElementById('latex-integral'), {
                    throwOnError: false,
                    displayMode: true
                });
            } catch (e) {
                // Fallback safe format
                document.getElementById('latex-integral').innerText = `Int[${limitA.toFixed(1)}, ${limitB.toFixed(1)}] = ${exact.signed.toFixed(4)}`;
            }

            requestRedraw();
        }

                katex.render(`f(x) = ${tex}`, document.getElementById('latex-formula'), {

        }

        // 2D CANVAS COORDINATE SYSTEM MAPPINGS
        function getCanvasWidth() { return canvas.width / (window.devicePixelRatio || 1); }
        function getCanvasHeight() { return canvas.height / (window.devicePixelRatio || 1); }

        function toScreenX(mathX) {
            const w = getCanvasWidth();
            return w / 2 + (mathX - centerX) * scaleX;
        }

        function toScreenY(mathY) {
            const h = getCanvasHeight();
            return h / 2 - (mathY - centerY) * scaleY;
        }

        function toMathX(screenX) {
            const w = getCanvasWidth();
            return centerX + (screenX - w / 2) / scaleX;
        }

        function toMathY(screenY) {
            const h = getCanvasHeight();
            return centerY - (screenY - h / 2) / scaleY;
        }

        // THE DRAW/RENDER GRAVITY ENGINE
        function drawGraph() {
            const width = getCanvasWidth();
            const height = getCanvasHeight();
            
            // Clear Frame
            ctx.fillStyle = COLORS.bg;
            ctx.fillRect(0, 0, width, height);
            
            // Update HUD View Range Indicator
            const mathMinX = toMathX(0);
            const mathMaxX = toMathX(width);
            document.getElementById('hud-scale-x').innerText = `[${mathMinX.toFixed(1)}, ${mathMaxX.toFixed(1)}]`;

            // Draw Dynamic Coordinate Grid Lines
            drawGrid(width, height);
            
            // Draw Main X and Y Axes
            drawAxes(width, height);

            // Shading Integration Area under function
            if (compiledExpr) {
                if (approxMethod === 'exact') {
                    shadeExactArea();
                } else {
                    shadeApproximationArea();
                }
                
                // Plot Function Curve
                drawCurve(width);
            }

            // Draw Limit Markers a & b
            drawLimits();

            // Hover Coordinate Inspector Crosshair
            if (mouseInCanvas && crosshairX !== null && compiledExpr) {
                drawCrosshair();
            }
        }

        // DRAW GRID LINES WITH COHERENT SCALING
        function drawGrid(width, height) {
            // Pick a dynamic scale based on scale pixels
            const roughStep = 60 / scaleX; // Target 60px intervals
            const magnitude = Math.pow(10, Math.floor(Math.log10(roughStep)));
            const ratio = roughStep / magnitude;
            
            let step = magnitude;
            if (ratio > 5) step = 5 * magnitude;
            else if (ratio > 2) step = 2 * magnitude;

            const mathLeft = toMathX(0);
            const mathRight = toMathX(width);
            const mathBottom = toMathY(height);
            const mathTop = toMathY(0);

            ctx.lineWidth = 1;
            ctx.textBaseline = 'middle';
            ctx.font = '10px Inter, sans-serif';

            // Vertical Grid Lines & Ticks on X
            const firstX = Math.floor(mathLeft / step) * step;
            for (let x = firstX; x <= mathRight; x += step) {
                const sx = toScreenX(x);
                
                // Grid line
                ctx.strokeStyle = COLORS.gridLines;
                ctx.beginPath();
                ctx.moveTo(sx, 0);
                ctx.lineTo(sx, height);
                ctx.stroke();

                // Draw Coordinate Number Label pinned to X axis or bottom
                let labelY = toScreenY(0) + 12;
                if (labelY < 12) labelY = 12;
                if (labelY > height - 15) labelY = height - 15;

                ctx.fillStyle = COLORS.text;
                ctx.textAlign = 'center';
                if (Math.abs(x) > 1e-9) {
                    ctx.fillText(x.toFixed(x % 1 === 0 ? 0 : 2), sx, labelY);
                }
            }

            // Horizontal Grid Lines & Ticks on Y
            const firstY = Math.floor(mathBottom / step) * step;
            for (let y = firstY; y <= mathTop; y += step) {
                const sy = toScreenY(y);

                ctx.strokeStyle = COLORS.gridLines;
                ctx.beginPath();
                ctx.moveTo(0, sy);
                ctx.lineTo(width, sy);
                ctx.stroke();

                // Draw Coordinate Number Label pinned to Y axis or side
                let labelX = toScreenX(0) - 10;
                if (labelX < 10) labelX = 10;
                if (labelX > width - 15) labelX = width - 15;

                ctx.fillStyle = COLORS.text;
                ctx.textAlign = 'right';
                if (Math.abs(y) > 1e-9) {
                    ctx.fillText(y.toFixed(y % 1 === 0 ? 0 : 2), labelX, sy);
                }
            }
        }

        // MAIN TRADITIONAL CARTESIAN AXES
        function drawAxes(width, height) {
            const sx0 = toScreenX(0);
            const sy0 = toScreenY(0);

            ctx.lineWidth = 2;
            ctx.strokeStyle = COLORS.axes;

            // X-Axis
            if (sy0 >= 0 && sy0 <= height) {
                ctx.beginPath();
                ctx.moveTo(0, sy0);
                ctx.lineTo(width, sy0);
                ctx.stroke();
            }

            // Y-Axis
            if (sx0 >= 0 && sx0 <= width) {
                ctx.beginPath();
                ctx.moveTo(sx0, 0);
                ctx.lineTo(sx0, height);
                ctx.stroke();
            }
        }

        // PERFECTLY FILL EXACT INTEGRATION AREA
        function shadeExactArea() {
            const startX = toScreenX(limitA);
            const endX = toScreenX(limitB);
            
            const minSX = Math.min(startX, endX);
            const maxSX = Math.max(startX, endX);
            const sy0 = toScreenY(0);

            // Integrate step-by-pixel column for maximum visual clarity
            ctx.lineWidth = 1;
            
            for (let sx = Math.ceil(minSX); sx <= Math.floor(maxSX); sx++) {
                const mx = toMathX(sx);
                const my = evalF(mx);
                if (isNaN(my) || !isFinite(my)) continue;

                const sy = toScreenY(my);

                // Set color depending on if value is positive or negative
                if (my >= 0) {
                    ctx.strokeStyle = COLORS.exactAreaPos;
                } else {
                    ctx.strokeStyle = COLORS.exactAreaNeg;
                }

                ctx.beginPath();
                ctx.moveTo(sx, sy0);
                ctx.lineTo(sx, sy);
                ctx.stroke();
            }

            // Highlight border lines at boundaries
            ctx.strokeStyle = 'rgba(255, 255, 255, 0.15)';
            ctx.lineWidth = 1.5;
            ctx.beginPath();
            ctx.moveTo(startX, sy0);
            ctx.lineTo(startX, toScreenY(evalF(limitA)));
            ctx.moveTo(endX, sy0);
            ctx.lineTo(endX, toScreenY(evalF(limitB)));
            ctx.stroke();
        }

        // RENDER APPROXIMATING SHADING BARRIER POLYS
        function shadeApproximationArea() {
            const h = (limitB - limitA) / subdivisionsN;
            const sy0 = toScreenY(0);

            for (let i = 0; i < subdivisionsN; i++) {
                const x_start = limitA + i * h;
                const x_end = limitA + (i + 1) * h;
                
                const sxStart = toScreenX(x_start);
                const sxEnd = toScreenX(x_end);
                
                if (approxMethod === 'trapezoid') {
                    // Trapezoidal requires polygons
                    const y1 = evalF(x_start);
                    const y2 = evalF(x_end);
                    if (!isFinite(y1) || !isFinite(y2)) continue;

                    const sy1 = toScreenY(y1);
                    const sy2 = toScreenY(y2);

                    const avgY = (y1 + y2) / 2;
                    ctx.fillStyle = avgY >= 0 ? COLORS.riemannAreaPos : COLORS.riemannAreaNeg;
                    ctx.strokeStyle = avgY >= 0 ? COLORS.riemannStrokePos : COLORS.riemannStrokeNeg;

                    ctx.beginPath();
                    ctx.moveTo(sxStart, sy0);
                    ctx.lineTo(sxStart, sy1);
                    ctx.lineTo(sxEnd, sy2);
                    ctx.lineTo(sxEnd, sy0);
                    ctx.closePath();
                    ctx.fill();
                    ctx.lineWidth = 1.5;
                    ctx.stroke();
                } else {
                    // Left, Right, or Midpoint Riemann Bars
                    let evalPoint = x_start;
                    if (approxMethod === 'right') evalPoint = x_end;
                    if (approxMethod === 'midpoint') evalPoint = x_start + h / 2;

                    const y = evalF(evalPoint);
                    if (isNaN(y) || !isFinite(y)) continue;

                    const sy = toScreenY(y);

                    ctx.fillStyle = y >= 0 ? COLORS.riemannAreaPos : COLORS.riemannAreaNeg;
                    ctx.strokeStyle = y >= 0 ? COLORS.riemannStrokePos : COLORS.riemannStrokeNeg;

                    // Draw the Bar
                    ctx.beginPath();
                    ctx.rect(sxStart, Math.min(sy0, sy), sxEnd - sxStart, Math.abs(sy0 - sy));
                    ctx.fill();
                    ctx.lineWidth = 1.5;
                    ctx.stroke();

                    // Optional dot at height sample point
                    const sxEval = toScreenX(evalPoint);
                    ctx.fillStyle = y >= 0 ? '#06b6d4' : '#f43f5e';
                    ctx.beginPath();
                    ctx.arc(sxEval, sy, 3, 0, 2 * Math.PI);
                    ctx.fill();
                }
            }
        }

        // PLOT HIGH-RESOLUTION FUNCTION CURVE
        function drawCurve(width) {
            ctx.save();
            ctx.lineWidth = 3.5;
            
            // Neon Glow Shadow Effect
            ctx.shadowColor = 'rgba(139, 92, 246, 0.4)';
            ctx.shadowBlur = 8;

            // Gradient Stroke for luxury visuals
            const gradient = ctx.createLinearGradient(0, 0, width, 0);
            gradient.addColorStop(0, COLORS.primaryCurve);
            gradient.addColorStop(1, COLORS.primaryCurveEnd);
            ctx.strokeStyle = gradient;

            ctx.beginPath();
            let isFirst = true;

            // Iterate over every horizontal pixel to map complex functions exactly
            for (let sx = 0; sx <= width; sx++) {
                const mx = toMathX(sx);
                const my = evalF(mx);

                if (isNaN(my) || !isFinite(my)) {
                    isFirst = true; // Break path on asymptotes / invalid values
                    continue;
                }

                const sy = toScreenY(my);

                if (isFirst) {
                    ctx.moveTo(sx, sy);
                    isFirst = false;
                } else {
                    // Handle high coordinates without breaking scaling limits
                    if (Math.abs(sy) < 50000) {
                        ctx.lineTo(sx, sy);
                    }
                }
            }

            ctx.stroke();
            ctx.restore();
        }

        // DRAGGABLE LIMIT INTERFACE ELEMENTS
        function drawLimits() {
            const sy0 = toScreenY(0);
            const h = getCanvasHeight();
            
            const limits = [
                { id: 'a', val: limitA, label: 'a', color: COLORS.limitA },
                { id: 'b', val: limitB, label: 'b', color: COLORS.limitB }
            ];

            limits.forEach(limit => {
                const sx = toScreenX(limit.val);

                // Dotted vertical limit line
                ctx.lineWidth = 1.5;
                ctx.strokeStyle = limit.color + '77'; // Semi-transparent
                ctx.setLineDash([6, 6]);
                ctx.beginPath();
                ctx.moveTo(sx, 0);
                ctx.lineTo(sx, h);
                ctx.stroke();
                ctx.setLineDash([]); // Reset dashed state

                // Glow hover state
                const isHovered = (hoverTarget === limit.id || dragTarget === limit.id);
                const baseRad = isHovered ? 12 : 9;

                // Handle boundary pin dot at X crossing point
                ctx.save();
                ctx.shadowColor = limit.color;
                ctx.shadowBlur = isHovered ? 12 : 5;

                // Outer Ring
                ctx.fillStyle = limit.color;
                ctx.beginPath();
                ctx.arc(sx, sy0, baseRad, 0, 2 * Math.PI);
                ctx.fill();

                // Inner core
                ctx.fillStyle = '#ffffff';
                ctx.beginPath();
                ctx.arc(sx, sy0, baseRad * 0.45, 0, 2 * Math.PI);
                ctx.fill();
                ctx.restore();

                // Drawing Flag/Tag directly above the axis dot
                const flagY = sy0 - 24;
                ctx.fillStyle = '#0f172a';
                ctx.strokeStyle = limit.color;
                ctx.lineWidth = 2;
                
                // Draw Pill Tag Shape
                roundRect(ctx, sx - 16, flagY - 10, 32, 20, 6, true, true);

                // Text centered inside pill
                ctx.fillStyle = '#ffffff';
                ctx.textAlign = 'center';
                ctx.textBaseline = 'middle';
                ctx.font = 'bold 11px Inter, sans-serif';
                ctx.fillText(limit.label, sx, flagY);
                
                // Print micro coordinate label floating above handle
                if (isHovered) {
                    ctx.fillStyle = '#e2e8f0';
                    ctx.font = '10px Inter, sans-serif';
                    ctx.fillText(limit.val.toFixed(2), sx, flagY - 18);
                }
            });
        }

        // TOOLTIP COORDINATE HOVER INDICATOR
        function drawCrosshair() {
            const sx = toScreenX(crosshairX);
            const sy = toScreenY(crosshairY);

            // Dotted crosshair line
            ctx.lineWidth = 1;
            ctx.strokeStyle = 'rgba(148, 163, 184, 0.25)';
            ctx.setLineDash([4, 4]);
            
            ctx.beginPath();
            ctx.moveTo(sx, 0);
            ctx.lineTo(sx, getCanvasHeight());
            ctx.moveTo(0, sy);
            ctx.lineTo(getCanvasWidth(), sy);
            ctx.stroke();
            
            ctx.setLineDash([]);

            // Intersecting Node Dot
            ctx.fillStyle = '#38bdf8';
            ctx.beginPath();
            ctx.arc(sx, sy, 5, 0, 2 * Math.PI);
            ctx.fill();

            // Label text coordinate box
            ctx.save();
            ctx.shadowColor = 'black';
            ctx.shadowBlur = 10;
            ctx.fillStyle = 'rgba(15, 23, 42, 0.95)';
            ctx.strokeStyle = '#38bdf8';
            ctx.lineWidth = 1.5;

            const text = `(${crosshairX.toFixed(3)}, ${crosshairY.toFixed(3)})`;
            ctx.font = '11px Inter, sans-serif';
            const padding = 8;
            const textWidth = ctx.measureText(text).width;
            
            const boxW = textWidth + padding * 2;
            const boxH = 24;
            const boxX = sx + 12 + boxW > getCanvasWidth() ? sx - 12 - boxW : sx + 12;
            const boxY = sy - 12;

            roundRect(ctx, boxX, boxY, boxW, boxH, 6, true, true);
            
            ctx.fillStyle = '#ffffff';
            ctx.textAlign = 'left';
            ctx.textBaseline = 'middle';
            ctx.fillText(text, boxX + padding, boxY + boxH / 2);
            ctx.restore();
        }

        // CANVAS ROUNDED RECTANGLE UTILITY
        function roundRect(ctx, x, y, width, height, radius, fill, stroke) {
            ctx.beginPath();
            ctx.moveTo(x + radius, y);
            ctx.lineTo(x + width - radius, y);
            ctx.quadraticCurveTo(x + width, y, x + width, y + radius);
            ctx.lineTo(x + width, y + height - radius);
            ctx.quadraticCurveTo(x + width, y + height, x + width - radius, y + height);
            ctx.lineTo(x + radius, y + height);
            ctx.quadraticCurveTo(x, y + height, x, y + height - radius);
            ctx.lineTo(x, y + radius);
            ctx.quadraticCurveTo(x, y, x + radius, y);
            ctx.closePath();
            if (fill) ctx.fill();
            if (stroke) ctx.stroke();
        }


        // INTERACTIVE POINTER EVENTS SYSTEM
        function getPointerPos(e) {
            const rect = canvas.getBoundingClientRect();
            // Accommodate mouse or touches
            const clientX = e.touches ? e.touches[0].clientX : e.clientX;
            const clientY = e.touches ? e.touches[0].clientY : e.clientY;
            return {
                x: clientX - rect.left,
                y: clientY - rect.top
            };
        }

        // Detect if pointer is near interactive handle dots on the X Axis
        function detectHandleHover(px, py) {
            const sy0 = toScreenY(0);
            
            const distA = Math.hypot(px - toScreenX(limitA), py - sy0);
            const distB = Math.hypot(px - toScreenX(limitB), py - sy0);

            // Radius threshold to grab (25 pixels)
            if (distA < 25 && distA < distB) {
                return 'a';
            } else if (distB < 25) {
                return 'b';
            }
            return null;
        }

        // Mouse Down / Touch Start
        function handlePointerDown(e) {
            const pos = getPointerPos(e);
            const hover = detectHandleHover(pos.x, pos.y);

            if (hover) {
                dragTarget = hover;
            } else {
                dragTarget = 'viewport'; // Start Panning view
            }

            lastMouseX = pos.x;
            lastMouseY = pos.y;
            
            // Prevent scrolling on mobile drag
            if (e.cancelable) e.preventDefault();
        }

        // Mouse Move / Touch Move
        function handlePointerMove(e) {
            const pos = getPointerPos(e);
            
            // Handle hover indicators when not dragging
            if (!dragTarget) {
                hoverTarget = detectHandleHover(pos.x, pos.y);
                if (hoverTarget) {
                    canvas.style.cursor = 'ew-resize';
                } else {
                    canvas.style.cursor = 'grab';
                }

                // Update Inspector Crosshair coords
                crosshairX = toMathX(pos.x);
                crosshairY = evalF(crosshairX);
                requestRedraw();
            } else {
                // Adjust drags
                if (dragTarget === 'a') {
                    let mx = toMathX(pos.x);
                    // Constraint limits within grid limits safely
                    limitA = Math.max(-20, Math.min(20, mx));
                    syncBoundsUI();
                    recalculateAndRender();
                } else if (dragTarget === 'b') {
                    let mx = toMathX(pos.x);
                    limitB = Math.max(-20, Math.min(20, mx));
                    syncBoundsUI();
                    recalculateAndRender();
                } else if (dragTarget === 'viewport') {
                    // Pan calculations
                    const dx = pos.x - lastMouseX;
                    const dy = pos.y - lastMouseY;

                    centerX -= dx / scaleX;
                    centerY += dy / scaleY;

                    lastMouseX = pos.x;
                    lastMouseY = pos.y;
                    
                    canvas.style.cursor = 'grabbing';
                    requestRedraw();
                }
            }

            if (e.cancelable) e.preventDefault();
        }

        // Mouse Up / Touch End
        function handlePointerUp() {
            dragTarget = null;
            canvas.style.cursor = hoverTarget ? 'ew-resize' : 'grab';
            requestRedraw();
        }

        // Connect Canvas Interaction Events
        canvas.addEventListener('mousedown', handlePointerDown);
        canvas.addEventListener('mousemove', handlePointerMove);
        window.addEventListener('mouseup', handlePointerUp);

        canvas.addEventListener('touchstart', handlePointerDown, { passive: false });
        canvas.addEventListener('touchmove', handlePointerMove, { passive: false });
        window.addEventListener('touchend', handlePointerUp);

        canvas.addEventListener('mouseenter', () => {
            mouseInCanvas = true;
        });

        canvas.addEventListener('mouseleave', () => {
            mouseInCanvas = false;
            crosshairX = null;
            crosshairY = null;
            requestRedraw();
        });

        // ZOOM CONTROLLER centered on pointer
        canvas.addEventListener('wheel', (e) => {
            e.preventDefault();
            const rect = canvas.getBoundingClientRect();
            const mouseX = e.clientX - rect.left;
            const mouseY = e.clientY - rect.top;

            const zoomFactor = e.deltaY < 0 ? 1.15 : 1 / 1.15;
            zoomAtCoordinate(mouseX, mouseY, zoomFactor);
        }, { passive: false });

        function zoomAtCoordinate(screenX, screenY, zoomFactor) {
            // Pick mathematical coordinate currently under cursor
            const mathX = toMathX(screenX);
            const mathY = toMathY(screenY);

            scaleX = Math.max(5, Math.min(2000, scaleX * zoomFactor));
            scaleY = Math.max(5, Math.min(2000, scaleY * zoomFactor));

            // Move viewport center so that math coord stays stationary in screen coordinates
            centerX = mathX - (screenX - getCanvasWidth() / 2) / scaleX;
            centerY = mathY + (screenY - getCanvasHeight() / 2) / scaleY;

            requestRedraw();
        }

        function zoom(factor) {
            const w = getCanvasWidth();
            const h = getCanvasHeight();
            zoomAtCoordinate(w / 2, h / 2, factor);
        }

        function resetViewport() {
            centerX = 0.0;
            centerY = 0.0;
            scaleX = 45.0;
            scaleY = 45.0;
            requestRedraw();
        }


        // UI BINDING INPUT ADJUSTERS
        function handleFunctionInput(value) {
            fExpression = value;
            if (compileExpression(fExpression)) {
                updateLatexFormulaDisplay();
                recalculateAndRender();
            }
        }

        function handleBoundSliderInput(bound, value) {
            const numeric = parseFloat(value);
            if (bound === 'a') {
                limitA = numeric;
                document.getElementById('input-a').value = numeric.toFixed(2);
            } else {
                limitB = numeric;
                document.getElementById('input-b').value = numeric.toFixed(2);
            }
            recalculateAndRender();
        }

        function handleBoundNumberInput(bound, value) {
            let numeric = parseFloat(value);
            if (isNaN(numeric)) return;

            // Expand slider boundary range if user specifies beyond bounds
            const slider = document.getElementById(`slider-${bound}`);
            const min = parseFloat(slider.min);
            const max = parseFloat(slider.max);

            if (numeric < min) {
                slider.min = Math.floor(numeric - 5);
            }
            if (numeric > max) {
                slider.max = Math.ceil(numeric + 5);
            }

            if (bound === 'a') {
                limitA = numeric;
                slider.value = numeric;
            } else {
                limitB = numeric;
                slider.value = numeric;
            }
            recalculateAndRender();
        }

        function adjustBound(bound, step) {
            if (bound === 'a') {
                limitA = parseFloat((limitA + step).toFixed(2));
            } else {
                limitB = parseFloat((limitB + step).toFixed(2));
            }
            syncBoundsUI();
            recalculateAndRender();
        }

        function swapLimits() {
            const temp = limitA;
            limitA = limitB;
            limitB = temp;
            syncBoundsUI();
            recalculateAndRender();
        }

        function handleIntervalSliderInput(value) {
            subdivisionsN = parseInt(value);
            document.getElementById('label-intervals').innerText = `n = ${subdivisionsN}`;
            recalculateAndRender();
        }

        // SYNCHRONIZE JS STATE TO DOM ELEMENTS
        function syncBoundsUI() {
            document.getElementById('input-a').value = limitA.toFixed(2);
            document.getElementById('slider-a').value = limitA;
            document.getElementById('input-b').value = limitB.toFixed(2);
            document.getElementById('slider-b').value = limitB;
        }

        // INTERACTIVE SWITCH BETWEEN MODE APPROXIMATIONS
        function setApproximationMethod(method) {
            approxMethod = method;
            
            const btnExact = document.getElementById('mode-exact');
            const btnRiemann = document.getElementById('mode-riemann');
            const riemannConfig = document.getElementById('riemann-config');

            // Set UI headers
            if (method === 'exact') {
                btnExact.className = "px-3 py-1.5 rounded-md font-medium transition-all duration-200 bg-brand-600 text-white";
                btnRiemann.className = "px-3 py-1.5 rounded-md font-medium transition-all duration-200 text-slate-400 hover:text-white";
                riemannConfig.classList.add('hidden');
            } else {
                btnExact.className = "px-3 py-1.5 rounded-md font-medium transition-all duration-200 text-slate-400 hover:text-white";
                btnRiemann.className = "px-3 py-1.5 rounded-md font-medium transition-all duration-200 bg-brand-600 text-white";
                riemannConfig.classList.remove('hidden');
            }

            // Highlight sub-type button inside config panel
            const approximationSubtypes = ['left', 'right', 'midpoint', 'trapezoid'];
            approximationSubtypes.forEach(type => {
                const button = document.getElementById(type === 'trapezoid' ? 'btn-trapezoid' : `btn-${type}-riemann`);
                if (button) {
                    if (method === type) {
                        button.classList.add('bg-brand-600', 'text-white', 'border-brand-500');
                        button.classList.remove('bg-slate-900', 'text-slate-300', 'border-slate-800');
                    } else {
                        button.classList.remove('bg-brand-600', 'text-white', 'border-brand-500');
                        button.classList.add('bg-slate-900', 'text-slate-300', 'border-slate-800');
                    }
                }
            });

            recalculateAndRender();
        }

            const data = PRESETS[key];
            if (!data) return;

            // Update inputs
            fExpression = data.expr;
            document.getElementById('function-input').value = data.expr;
            
            limitA = data.a;
            limitB = data.b;
            syncBoundsUI();

            // Refocus graph camera context
            centerX = data.cx;
            centerY = data.cy;
            scaleX = data.sc;
            scaleY = data.sc;

            // Re-render
            compileExpression(fExpression);
            updateLatexFormulaDisplay();
            recalculateAndRender();
        }
